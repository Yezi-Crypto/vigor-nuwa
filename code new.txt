    <script>
        // Initialize Lucide Icons
        lucide.createIcons();

        // Sticky Navbar Logic
        const navbar = document.getElementById('navbar');
        
        window.addEventListener('scroll', () => {
            if (window.scrollY > 50) {
                navbar.classList.add('bg-paper', 'text-tea-brown', 'shadow-md', 'py-2');
                navbar.classList.remove('bg-transparent', 'text-white', 'py-4');
                // Change logo icon color on scroll
                navbar.querySelectorAll('[data-lucide]').forEach(icon => {
                    if(icon.getAttribute('data-lucide') === 'leaf') {
                        icon.classList.remove('text-glazed');
                        icon.classList.add('text-grass-green');
                    }
                });
            } else {
                navbar.classList.remove('bg-paper', 'text-tea-brown', 'shadow-md', 'py-2');
                navbar.classList.add('bg-transparent', 'text-white', 'py-4');
                // Revert logo icon color
                navbar.querySelectorAll('[data-lucide]').forEach(icon => {
                    if(icon.getAttribute('data-lucide') === 'leaf') {
                        icon.classList.add('text-glazed');
                        icon.classList.remove('text-grass-green');
                    }
                });
            }
        });

        // Scroll Reveal Animation with Intersection Observer
        const observerOptions = {
            root: null,
            rootMargin: '0px',
            threshold: 0.15
        };

        const observer = new IntersectionObserver((entries, observer) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('active');
                    observer.unobserve(entry.target); // Run once
                }
            });
        }, observerOptions);

        document.querySelectorAll('.reveal').forEach((element) => {
            observer.observe(element);
        });

        // Mobile Menu Toggle (Basic implementation for demo)
        document.getElementById('mobile-menu-btn').addEventListener('click', () => {
            alert('Mobile menu would open here. (Can be implemented with a simple fixed div overlay)');
        });

        // --- Gemini API Integration ---
        const apiKey = ""; 
        const generateBtn = document.getElementById('generate-btn');
        const userInput = document.getElementById('user-symptoms');
        const loadingDiv = document.getElementById('ai-loading');
        const resultDiv = document.getElementById('ai-result');
        const responseText = document.getElementById('ai-response-text');

        async function getHerbalRecommendation(prompt) {
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
            
            // System prompt configuring Gemini's persona and output style
            const systemPrompt = `You are a wise Royal Physician from the Northern Song Dynasty specializing in Traditional Chinese Medicine and 'Medicine and Food Homology'. 
            The user will tell you their symptoms, feelings, or available ingredients. 
            Provide a short, compassionate, and practical dietary or herbal recommendation to balance their Yin and Yang. 
            Keep the tone poetic, ancient, and deeply healing. 
            Format the response using basic HTML tags (like <strong> for emphasis, <br> for line breaks, or <ul><li> for lists) so it looks good on a webpage. 
            Do NOT wrap the response in markdown blocks. Keep it under 150 words.`;

            const payload = {
                contents: [{ parts: [{ text: prompt }] }],
                systemInstruction: { parts: [{ text: systemPrompt }] }
            };

            let retries = 5;
            let delay = 1000;
            
            while (retries > 0) {
                try {
                    const response = await fetch(url, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });
                    
                    if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);
                    
                    const result = await response.json();
                    return result.candidates?.[0]?.content?.parts?.[0]?.text || "The scroll is blank. Please try asking again.";
                } catch (error) {
                    retries--;
                    if (retries === 0) {
                        return "<em>The connection to the ancient scrolls has faded. Please try again later.</em>";
                    }
                    await new Promise(res => setTimeout(res, delay));
                    delay *= 2;
                }
            }
        }

        generateBtn.addEventListener('click', async () => {
            const prompt = userInput.value.trim();
            if (!prompt) return;

            // Update UI for loading state
            generateBtn.disabled = true;
            generateBtn.classList.add('opacity-70', 'cursor-not-allowed');
            resultDiv.classList.add('hidden');
            loadingDiv.classList.remove('hidden');

            // Call Gemini API
            const answerHTML = await getHerbalRecommendation(prompt);

            // Show Result
            loadingDiv.classList.add('hidden');
            responseText.innerHTML = answerHTML;
            resultDiv.classList.remove('hidden');
            
            // Restore Button
            generateBtn.disabled = false;
            generateBtn.classList.remove('opacity-70', 'cursor-not-allowed');
        });
    </script>
